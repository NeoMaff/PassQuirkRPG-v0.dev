# Discord Bot Configuration
# Copy this file to .env and fill in the values

# Bot Token (DISCORD_TOKEN=)
# Get this from the Discord Developer Portal: https://discord.com/developers/applications
DISCORD_TOKEN=your_bot_token_here

# Application ID (CLIENT_ID=)
# Also from the Discord Developer Portal
CLIENT_ID=your_client_id_here

# Server ID (GUILD_ID=) - Optional, for development
# Only register commands to a specific guild for testing
GUILD_ID=your_guild_id_here

# Bot Owner ID (OWNER_ID=) - Optional
# Used for owner-only commands
OWNER_ID=your_discord_id_here

# Bot Prefix (PREFIX=) - Optional, default is '!'
PREFIX=!

# Database Configuration
# SQLite is used by default (no configuration needed)
# For other databases, uncomment and configure below
# DATABASE_URL=postgres://user:password@localhost:5432/database
# DB_DIALECT=postgres
# DB_STORAGE=./data/database.sqlite

# Logging
LOG_LEVEL=info
LOG_TO_FILE=true
LOG_TO_CONSOLE=true

# Development Settings
NODE_ENV=development
DEBUG=true
SYNC_DATABASE=false
RESET_ON_START=false

# Panel Configuration
# Enable/disable specific panels
ENABLE_BATTLE=true
ENABLE_INVENTORY=true
ENABLE_CHARACTER_CREATION=true
ENABLE_DUNGEON=true
ENABLE_TOURNAMENT=true
ENABLE_ECONOMY=true
ENABLE_LEVELING=true
